# 02장. 파이썬 기초 익히기

파이썬은 1991년 귀도 반 로섬이 발표한 고급 프로그래밍 언어로 문법이 간단하고 라이브러리가 풍부하다. 인터프리터식, 객체지향적, 동적 타이핑 대화형 언어로서 백엔드와 머신러닝 분야를 비롯한 다양한 영역에서 사용된다.  

```python
a = 3
b = 2

# 산술 연산: +, -, * 등
a + b   # 더하기
a - b   # 빼기
a * b   # 곱하기
a ** b  # 거듭제곱
a / b   # 나누기
a // b  # 나누기 연산 후 소수점 이하를 버린 정수(몫)
a % b   # 나누기 연산 후 나머지


# 출력 함수: print()
print('Hello World!')


# 자료형과 자료구조
## 숫자형: 정수형, 실수형, 복소수형
1      # 정수형
3.14   # 실수형: 소수점이 있는 숫자
2 + 5j # 복소수형: 실수와 허수로 이루어진 숫자

## 논리형(Boolean)
True   # 참
False  # 거짓

## 군집형
[1, 2, 3] # 리스트: 순서가 있는 데이터의 모음
(4, 5, 6) # 튜플: 순서가 있는 데이터의 모음, 한 번 정의되면 수정이 불가능
{ 'name': '홍길동', 'age': 20 } # 딕셔너리: 키와 값으로 구성된 자료구조
{1, 3, 5} # 중복된 값을 보유하지 않는 자료구조, 순서가 없다.
'hello' # 문자열


# 반복문: for, while
for i in [1, 2, 3]:
    print(i)

a = 3
while a > 0:
    print(a)

# 조건문: if
a = 3
if i > 0:
    print('양수')
elif i < 0:
    print('음수')
else:
    print('0')


# 파이썬 내장 함수: abs(), len() 등
abs(-3) # 절댓값: 3
len([1,2,3,4,5]) # 데이터의 길이: 5
max([1,2,3,4,5]) # 최댓값: 5
min([1,2,3,4,5]) # 최솟값: 1
list(range(1,5)) # range(): 데이터의 범위 지정
round(3.8) # 반올림: 4
sum([1,2,3]) # 합계: 6


# 함수 만들기: def
## 기본 함수
def plus_ten(x):
    return x + 10

plus_ten(5) # 15

## 기본값이 있는 함수: 매개변수가 없으면 기본값으로 설정된다.
def sum_ten(a, b=10):
    return a + b

sum_ten(5) # 15
```
